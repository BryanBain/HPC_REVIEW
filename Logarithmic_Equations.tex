\chapter{Logarithmic Equations}

Solve each. Round to 3 decimal places when necessary.
\begin{enumerate}
	\item $\log_5(x) + x\log_5(x) > 0$
	\item $\ln\left(8-x^2\right) = \ln(2-x)$
	\item $\log_{25}\left(\frac{3x+1}{2x-2}\right) = \frac{1}{2}$
\end{enumerate}
