\begin{enumerate}
	\item $b(x) = \log_7\left(x^2 - 8x + 6\right)$
	\item $a(x) = \ln\left(\frac{x^2+3x+2}{5x+15}\right)$
	\item $f(x) = -7\ln\left(x^2 + 9x + 8\right)$
	\item $g(x) = \log\left(5x^2 + 13x - 6\right)$
	\item $h(x) = 3\log_2\left(x^3+2x^2-x-2\right)$
\end{enumerate}
