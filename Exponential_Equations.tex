\chapter{Exponential Equations}

Solve each. Round to 3 decimal places when necessary.
\begin{enumerate}
	\item $3e^{x-2} = 7$
	\item $5^x + 4 > 1$
	\item $2^{3x+4} = 32^{x-7}$
	\item $5e^{7x} + 10 = 42$
	\item $7^{4x+1} \geq 343$
	\item $1000e^{0.04x} = 2000$
	\item $3(4.1)^{x-2} = 8$
	\item $2^{x+1} = 5^{7x-5}$
\end{enumerate}
